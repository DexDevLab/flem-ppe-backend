    import axios from "axios";
import { exceptionHandler } from "utils/exceptionHandler";

/**
 * Controller que lista os Territórios de Identidade providos pela SEAGRI-BA.
 * Consulta uma API externa para prover os dados desejados. Este controller
 * é utilizado para testes; o de Produção deve utilizar uma consulta via BD.
 * @method fetchTerritoriosIdentidade
 * @memberof module:ext
 * @returns {Object} Objeto contendo informações do resultado.
 */
export async function fetchTerritoriosIdentidade() {
  try {
    const url = `${process.env.NEXT_PUBLIC_API_TERRITORIOS_BA}`;
    const fetch = await axios.get(url);
    const data = fetch.data;
    const values = data.drawingInfo.renderer.uniqueValueInfos;
    const territorios = new Array();
    values.forEach(async (item, idx) => {
      territorios.push({
        id: idx,
        territorio: item.label,
        municipios: await getMunicByTerritorio(item.label),
      });
    });
    return territorios;
  } catch (e) {
    throw exceptionHandler(e, 0);
  }
}

/**
 * Fornece a listagem de municípios por território de identidade.
 * @method getMunicByTerritorio
 * @memberof module:territorios-identidade
 * @param {String} territ o território de identidade
 * @returns {Object} Objeto contendo um Array de resultados.
 */
export async function getMunicByTerritorio(territ) {
  const munic = [];
  listaTerritoriosIdentidade.map((item) => {
    if (item.territorio == territ) {
      munic.push(item.municipios);
    }
  });
  return munic[0];
}

/**
 * Fornece a listagem de territórios por município.
 * @method getTerritorioByMunic
 * @memberof module:ext
 * @param {String} munic o município
 * @returns {Object} Objeto contendo um Array de resultados.
 */
export async function getTerritorioByMunic(munic) {
  const territ = [];
  listaTerritoriosIdentidade.forEach((item) => {
    item.municipios.forEach((item2) => {
      if (item2 === munic) {
        territ.push(item.territorio);
      }
    });
  });
  return territ[0];
}
